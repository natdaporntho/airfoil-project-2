36   def read_xflr5(filename):
37       """ 
38       Reads XFLR5 aerodynamic data and processes key coefficients. 
39    
40       Parameters: 
41           filename (str): Path to the .csv file exported from XFLR5. 
42    
43       Returns: 
44           pandas.DataFrame: Filtered aerodynamic data with additional Cl/Cd ratio. 
45       """
46       df = pd.read_csv(
47           filename,
48           skiprows=10,
49           names=["alpha", "CL", "CD", "CDp", "Cm", "Top Xtr", "Bot Xtr", "Cpmin", "Chinge", "XCp"]
50       )
51   
52       # Filter angles between 0 and 12 degrees, even numbers only
53       df_filtered = df[
54           (df['alpha'].between(0, 12)) &
55           (df['alpha'] % 2 == 0)
56           ].copy()
57   
58       # Calculate Lift-to-Drag Ratio
59       df_filtered["CL/CD"] = df_filtered["CL"] / df_filtered["CD"]
60   
61       return df_filtered

97   def get_lift_curve_slope_xflr5(airfoil_code, reynolds_number):
98       filename = f'NACA {airfoil_code}_T1_Re0.{str(reynolds_number)[:3]}_M0.03_N9.0.csv'
99       data = read_xflr5(filename)
100  
101      alpha_max_cl = get_alpha_at_max_cl_xflr5(airfoil_code, reynolds_number)
102  
103      cl_before = data.loc[data['alpha'] == alpha_max_cl - 2.0, 'CL'].values[0]
104      cl_at_2deg = data.loc[data['alpha'] == 2.0, 'CL'].values[0]
105  
106      slope = (cl_before - cl_at_2deg) / ((alpha_max_cl - 4.0) * math.pi / 180)
107      return slope

130  def drag_coefficient(drag_force, velocity):
131      return 2 * drag_force / (1.2 * 0.3 * 0.15 * velocity ** 2)
132  
133  def lift_coefficient(lift_force, velocity):
134      return 2 * lift_force / (1.2 * 0.3 * 0.15 * velocity ** 2)

136  def clean_experiment_data(airfoil_code, test_velocity):
137      airfoil_code = str(airfoil_code)
138      file_name = f'{airfoil_code} v={test_velocity}.csv'
139  
140      data = pd.read_csv(file_name)
141      data.columns = data.iloc[0]
142      data = data[1:].reset_index(drop=True)
143  
144      data = data.iloc[:, :7]
145      data = data.drop(columns=['Time\n(h:min:sec)'])
146  
147      data['Act.AirSpeed\n(m/s)'] = data['Act.AirSpeed\n(m/s)'].astype(float)
148      mean_air_speed = data['Act.AirSpeed\n(m/s)'].mode()[0]
149  
150      data = data[(data['Act.AirSpeed\n(m/s)'] < mean_air_speed + 0.5) &
151                  (data['Act.AirSpeed\n(m/s)'] > mean_air_speed - 0.5)]
152  
153      data['DragForce\n(N)'] = data['DragForce\n(N)'].astype(float) * (-1)
154      data['LiftForce\n(N)'] = data['LiftForce\n(N)'].astype(float)
155  
156      data['LD'] = data['LiftForce\n(N)'] / data['DragForce\n(N)']
157      data['Act.Angle\n(deg)'] = data['Act.Angle\n(deg)'].astype(float)
158  
159      # Filter for even angles of attack below 14 degrees
160      data = data[data['Act.Angle\n(deg)'] % 2 == 0]
161      data = data[data['Act.Angle\n(deg)'] < 14]
162  
163      data['aoa'] = data['Act.Angle\n(deg)']  # Angle of attack
164  
165      data['CD'] = round(drag_coefficient(data['DragForce\n(N)'], test_velocity), 6)
166      data['CL'] = round(lift_coefficient(data['LiftForce\n(N)'], test_velocity), 6)
167  
168      data['lift_force'] = data['LiftForce\n(N)']
169      data['lift_to_drag_ratio'] = data['LD']
170  
171      return data

173  def group_coefficient(coef_type, airfoil_code, test_velocity):
174      data = clean_experiment_data(airfoil_code, test_velocity)
175      return data.groupby('aoa')[coef_type].apply(lambda x: x.mode()[0]).reset_index()

181      ex = clean_experiment_data(model, v)

810  def pearson(foil, what):
811      global m1
812      m2 = read_xflr5(f'NACA {foil}_T1_Re0.119_M0.03_N9.0.csv')
813      if what == 'LD':
814          m1 = group_coefficient('LD', foil, test_velocity = 12)
815          merged = pd.merge(m1, m2, left_on='aoa', right_on = 'alpha', how='inner')
816          m1 = merged['LD']
817          m2 = merged['CL/CD']
818  
819      elif what == 'L':
820          m1 = group_coefficient('CL', foil, test_velocity = 12)
821          m1 = m1.rename(columns={'CL': 'LL'})
822          merged = pd.merge(m1, m2, left_on='aoa', right_on = 'alpha', how='inner')
823          m1 = merged['LL']
824          m2 = merged['CL']
825  
826      elif what == 'D':
827          m1 = group_coefficient('CD', foil, test_velocity = 12)
828          m1 = m1.rename(columns={'CD': 'DD'})
829          merged = pd.merge(m1, m2, left_on='aoa', right_on = 'alpha', how='inner')
830          m1 = merged['DD']
831          m2 = merged['CD']
832  
833      method1 = np.array(m1)
834      method2 = np.array(m2)
835      r, p = pearsonr(method1, method2)
836      return r
